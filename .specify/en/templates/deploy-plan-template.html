Database Migrations

1
Create user preferences table
UP
CREATE TABLE user_preferences (
  id SERIAL PRIMARY KEY,
  user_id INTEGER REFERENCES users(id),
  preference_key VARCHAR(255) NOT NULL,
  preference_value JSONB,
  created_at TIMESTAMP DEFAULT NOW()
);
DOWN
DROP TABLE IF EXISTS user_preferences;
2
Add notification settings columns
UP
ALTER TABLE users
	ADD COLUMN notification_enabled BOOLEAN DEFAULT true,
	ADD COLUMN notification_frequency VARCHAR(20) DEFAULT 'daily';
DOWN
ALTER TABLE users
	DROP COLUMN notification_enabled,
	DROP COLUMN notification_frequency;

Initialization Scripts

bash 
#!/bin/bash
set -euo pipefail

echo "==> Running database migrations..."
npx db-migrate up

echo "==> Seeding initial data..."
node scripts/seed.js

echo "==> Building static assets..."
npm run build

echo "==> Running health check..."
curl -f http://localhost:8080/health || exit 1

echo "==> Deployment complete!"
bash 
#!/bin/bash
# Rollback script
set -euo pipefail

echo "==> Rolling back database migrations..."
npx db-migrate down

echo "==> Restoring previous build..."
cp -r .backup/latest ./dist

echo "==> Restarting service..."
pm2 restart app --update-env

echo "==> Rollback complete!"
